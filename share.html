        function useFirebase() {

            const [db, setDb] = useState(null);

            const [userId, setUserId] = useState(null);

            const [isReady, setIsReady] = useState(false);

                    const checkDependencies = setInterval(() => {

                        if (window.firebaseLoaded && window.firebaseDependencies) {

                            clearInterval(checkDependencies);

                            initialize();

                        }

                    }, 100);

                    return () => clearInterval(checkDependencies);

                } else {

                    initialize();

                }


                function initialize() {

                    const {

                        initializeApp, getAuth, signInAnonymously, signInWithCustomToken, onAuthStateChanged,

                        getFirestore, setLogLevel

                    } = window.firebaseDependencies;


                    setLogLevel('debug');


                    if (!firebaseConfig || !Object.keys(firebaseConfig).length) {

                        console.error("Firebase config is missing.");

                        setIsReady(true);

                        return;

                    }


                    try {

                        const app = initializeApp(firebaseConfig);

                        const authInstance = getAuth(app);

                        const dbInstance = getFirestore(app);


                        setDb(dbInstance);


                        const authenticate = async () => {

                            try {

                                if (initialAuthToken) {

                                    await signInWithCustomToken(authInstance, initialAuthToken);

                                } else {

                                    await signInAnonymously(authInstance);

                                }

                            } catch (error) {

                                console.error("Firebase Sign-in/Auth Error:", error);

                            }

                        };


                        authenticate();


                        const unsubscribe = onAuthStateChanged(authInstance, (user) => {

                            if (user) {

                                setUserId(user.uid); 

                            } else {

                                setUserId(crypto.randomUUID()); 

                            }

                            setIsReady(true);

                        });


                        return () => unsubscribe();

                    } catch (e) {

                        console.error("Firebase Initialization Critical Error:", e);

                        setIsReady(true);

                    }

                }

            }, []);


            return { db, userId, isReady };

        }

        // 개별 과목 방 컴포넌트

        const SubjectRoom = ({ subject, db, userId, nickname, onBack }) => {

            const [posts, setPosts] = useState([]);

            const [title, setTitle] = useState('');

            const [content, setContent] = useState('');

            const [imageFile, setImageFile] = useState(null); 

            const [imagePreview, setImagePreview] = useState(null); 

            const [isLoading, setIsLoading] = useState(true);

            const [uploadError, setUploadError] = useState('');

            const [showErrorModal, setShowErrorModal] = useState(false);

            const [modalMessage, setModalMessage] = useState("");


            const collectionPath = `artifacts/${appId}/public/data/exam_posts`;


            // 커스텀 모달 표시 함수 (alert 대체)

            const showCustomAlert = (message) => {

                setModalMessage(message);

                setShowErrorModal(true);

            };


            // 이미지 파일 핸들러 (Base64 변환)

            const handleFileChange = (e) => {

                const file = e.target.files[0];

                setUploadError(''); 

                if (!file) {

                    setImageFile(null);

                    setImagePreview(null);

                    return;

                }


                if (file.size > MAX_FILE_SIZE_BYTES) {

                    setUploadError(`파일 크기가 ${MAX_FILE_SIZE_KB}KB를 초과합니다. 더 작은 이미지를 선택해주세요.`);

                    setImageFile(null);

                    setImagePreview(null);

                    return;

                }


                const reader = new FileReader();

                reader.onloadend = () => {

                    setImageFile(reader.result);

                    setImagePreview(reader.result);

                };

                reader.onerror = () => {

                    setUploadError("파일을 읽는 도중 오류가 발생했습니다.");

                    setImageFile(null);

                    setImagePreview(null);

                };

                reader.readAsDataURL(file); 

            };


            // 파일 선택 취소 핸들러

            const handleCancelImage = () => {

                setImageFile(null);

                setImagePreview(null);

                setUploadError('');

                const fileInput = document.getElementById('file-upload');

                if (fileInput) fileInput.value = '';

            };


            // 실시간 데이터 리스너

            useEffect(() => {

                if (!db || !window.firebaseDependencies) return; 


                const { collection, query, where, onSnapshot } = window.firebaseDependencies;

                

                // Firestore 인덱스 오류 방지 및 JS 정렬을 위해 orderBy는 사용하지 않습니다.

                const q = query(

                    collection(db, collectionPath),

                    where("subject", "==", subject)

                );


                const unsubscribe = onSnapshot(q, (snapshot) => {

                    let fetchedPosts = snapshot.docs.map(doc => ({

                        id: doc.id,

                        ...doc.data()

                    }));

                    

                    // JS에서 정렬 (최신순)

                    fetchedPosts.sort((a, b) => {

                        if (!a.timestamp) return -1;

                        if (!b.timestamp) return 1;

                        const dateA = a.timestamp && typeof a.timestamp.toDate === 'function' ? a.timestamp.toDate() : new Date(0);

                        const dateB = b.timestamp && typeof b.timestamp.toDate === 'function' ? b.timestamp.toDate() : new Date(0);

                        return dateB - dateA;

                    });


                    setPosts(fetchedPosts);

                    setIsLoading(false);

                }, (error) => {

                    console.error("Error fetching posts:", error);

                    setIsLoading(false);

                    showCustomAlert("정보를 불러오는 중 오류가 발생했습니다. 잠시 후 다시 시도해 주세요.");

                });


                return () => unsubscribe();

            }, [db, subject]);


            // 게시물 추가 핸들러

            const handleSubmit = async (e) => {

                e.preventDefault();

                if (!db || !window.firebaseDependencies) return;


                const { collection, addDoc, serverTimestamp } = window.firebaseDependencies;


                if (!nickname || !title.trim() || !content.trim()) {

                    showCustomAlert("닉네임, 제목, 내용을 모두 입력해주세요.");

                    return;

                }


                if (uploadError) {

                    showCustomAlert("이미지 파일 오류를 해결한 후 다시 시도해주세요.");

                    return;

                }


                try {

                    await addDoc(collection(db, collectionPath), {

                        subject: subject,

                        title: title.trim(),

                        content: content.trim(),

                        imageDataUrl: imageFile || null, 

                        nickname: nickname, 

                        timestamp: serverTimestamp(),

                    });


                    // 폼 초기화

                    setTitle('');

                    setContent('');

                    handleCancelImage(); 


                } catch (error) {

                    console.error("Error adding document: ", error);

                    showCustomAlert("게시물 등록 중 오류가 발생했습니다. (자세한 내용은 콘솔 확인)");

                }

            };


            const formatDate = (timestamp) => {

                if (!timestamp) return "등록 중...";

                if (typeof timestamp.toDate !== 'function') return "업로드 대기 중..."; 

                const date = timestamp.toDate();

                return date.toLocaleString('ko-KR', {

                    month: '2-digit', day: '2-digit',

                    hour: '2-digit', minute: '2-digit'

                });

            };
